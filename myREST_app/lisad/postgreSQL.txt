CREATE SEQUENCE book_id ;
CREATE TABLE book
(
   id               numeric (10, 0) NOT NULL DEFAULT nextval('book_id'),
   book_name        text,
   book_author      text,
   pages_total      numeric (5, 0),
   print_out_year   numeric (4, 0),
   CONSTRAINT book_pk PRIMARY KEY (id)
)
